cmake_minimum_required(VERSION 3.10)
project(Social-Networking-App)

set(CMAKE_CXX_STANDARD 20)

add_definitions(-DSFML_STATIC)


set(SFML_DIR "") # insert path
set(SFML_INCLUDE_DIR "${SFML_DIR}/include")
set(SFML_LIBRARY_DIR "${SFML_DIR}/lib")

include_directories(${SFML_INCLUDE_DIR})
link_directories(${SFML_LIBRARY_DIR})


set(SFML_EXTERNAL_DEPS
    vorbisfile
    vorbisenc
    vorbis
    ogg
    FLAC
    freetype
)


if (WIN32)
    set(SFML_SYSTEM_DEPS
        ws2_32
        opengl32
        winmm
        gdi32
    )
endif()


add_executable(Social-Networking-App main.cpp) #add win32 in the middle if no-console

target_link_libraries(Social-Networking-App
    sfml-graphics-s
    sfml-window-s
    sfml-audio-s
    sfml-system-s
    
    
    ${SFML_EXTERNAL_DEPS}
    
    
    ${SFML_SYSTEM_DEPS}
)
